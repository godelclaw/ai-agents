\begin{abstract}
This note documents the current theorem-level state of the $\pi$-to-$\rho$ embedding formalization in Mettapedia, with explicit separation between canonical core $\rho$ semantics and a derived administrative layer. It records completed forward/admin and backward/partition endpoints, the canonical packaged API exposed from \texttt{PiCalculus/Main.lean}, and cleanup actions performed in this phase. The intent is to provide an auditable checkpoint before the OSLF-level completion stage.
\paragraph{Status note.} Implementation-status and count claims in this draft are snapshot-scoped; verify against current repository trackers and build outputs before citing as current state. Lean-proven claims are only those tied to explicit Lean file/theorem references; broader mathematical or historical claims are literature-backed context and not asserted as Lean-proven unless explicitly stated.
\end{abstract}

\section{Semantic Layers and Contract}
The implementation is explicitly two-layered:
\begin{itemize}
  \item \textbf{Core $\rho$ semantics} (canonical): one-step reduction \texttt{Reduces} and multistep \texttt{ReducesStar} in
  \texttt{Mettapedia/Languages/ProcessCalculi/RhoCalculus/Reduction.lean} and
  \texttt{.../MultiStep.lean}.
  \item \textbf{Derived/admin semantics}: \texttt{ReducesDerived} and \texttt{ReducesDerivedStar} in
  \texttt{Mettapedia/Languages/ProcessCalculi/RhoCalculus/DerivedRepNu.lean}, including replication unfold and context closure.
\end{itemize}

Derived bisimilarity is formalized separately as:
\begin{itemize}
  \item \texttt{WeakRestrictedBisimD} in
  \texttt{Mettapedia/Languages/ProcessCalculi/PiCalculus/WeakBisimDerived.lean}.
\end{itemize}

\subsection*{Listing: Derived bisimulation contract}
\begin{lstlisting}
-- Mettapedia/Languages/ProcessCalculi/PiCalculus/WeakBisimDerived.lean
def WeakRestrictedBisimD (N : Finset String) (p q : Pattern) : Prop :=
  Exists (fun R : Pattern -> Pattern -> Prop =>
    (forall p1 q1, R p1 q1 -> R q1 p1) /\
    (forall p1 q1, R p1 q1 -> forall p2, Nonempty (ReducesDerived p1 p2) ->
      Exists (fun q2 => Nonempty (ReducesDerivedStar q1 q2) /\ R p2 q2)) /\
    (forall p1 q1, R p1 q1 -> forall x, x in N ->
      RhoObservableSC p1 (.fvar x) -> RhoObservableSC q1 (.fvar x)) /\
    R p q
  )
\end{lstlisting}

inductive EncodedSCStepSource (N : Finset String) : Pattern -> Prop where
  | admin {src : Pattern} :
      AdminSource src ->
      EncodedSCStepSource N src
  | rfStepAt {src : Pattern}
      (P Q : Process) (n v : String)
      (h : ForwardSimulation.ReducesRF P Q)
      (hrf : ForwardSimulation.RestrictionFree P)
      (hsafe : ForwardSimulation.CommSafe h)
      (hsrcSC : StructuralCongruence src (encode P n v)) :
      EncodedSCStepSource N src

theorem weak_backward_reflection_step_encodedSC_partition_target_sensitive
    {N : Finset String} {src tgt : Pattern}
    (hsrc : EncodedSCStepSource N src)
    (hstep : Nonempty (ReducesDerived src tgt)) :
    (Exists (fun tgt' => Exists (fun canon =>
      Nonempty (ReducesDerivedStar src tgt') /\
      AdminCanonicalTarget src canon /\
      WeakRestrictedBisimD N tgt' canon)))
    \/
    (Nonempty (Reduction.Reduces src tgt) /\
      Exists (fun P => Exists (fun P' => Exists (fun n => Exists (fun v => Exists (fun tgt' =>
        StructuralCongruence src (encode P n v) /\
        Nonempty (PiReduction.Reduces P P') /\
        Nonempty (ReducesDerivedStar (encode P n v) tgt') /\
        WeakRestrictedBisimD N tgt' (encode P' n v)
      ))))))
\end{lstlisting}

\begin{itemize}
